--> to tackle overloading, we have to used different parameter in same name function.
     EX: 
        myfunc(int a)
        {
            cout << "This is Function 1." << endl;
        }
        myfunc(string a)
        {
            cout << "This is Function 2." << endl;
        }
        myfunc(int a, int b)
        {
            cout << "This is Function 3." << endl;
        }
        myfunc()
        {
            cout << "This is Function 4." << endl;
        }
